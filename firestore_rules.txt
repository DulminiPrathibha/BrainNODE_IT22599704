// Firebase Firestore Security Rules for BrainNODE
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users collection
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Quizzes collection - teachers can write, students can read published quizzes
    match /quizzes/{quizId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && 
        (resource == null || resource.data.teacherId == request.auth.uid);
    }
    
    // Quiz attempts - students can write their own attempts, teachers can read all
    match /quiz_attempts/{attemptId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && 
        (resource == null || resource.data.studentId == request.auth.uid);
    }
    
    // Mistake cards - students can read/write their own mistake cards
    match /mistake_cards/{mistakeCardId} {
      allow read, write: if request.auth != null && 
        (resource == null || resource.data.studentId == request.auth.uid);
    }
    
    // Notes collection
    match /notes/{noteId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && 
        (resource == null || resource.data.teacherId == request.auth.uid);
    }
    
    // Student progress
    match /student_progress/{progressId} {
      allow read, write: if request.auth != null && 
        (resource == null || resource.data.studentId == request.auth.uid);
    }
  }
}
